function supportsSVG() {
        return document.implementation.hasFeature( "http://www.w3.org/TR/SVG11/feature#Image", "1.1" );
    }

    if( !supportsSVG() ) {
        $( 'img' ).each( function() {
            var src = $( this ).attr( 'src' );
            if( src.indexOf( '.svg' ) !== -1 ) {
                newSrc = src.replace('.svg', '.png');
                $(this).attr('src', newSrc);
            }
        } );
    }
